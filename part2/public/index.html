const { createApp } = Vue; createApp({ data() { return { message: 'Welcome to the Dog Walking Service!', isLoggedIn: false, currentUser: null, isLoading: false, loginError: '', loginForm: { email: '', password: '' } }; }, async mounted() { await this.checkLoginStatus(); }, methods: { async checkLoginStatus() { try { const response = await fetch('/api/users/me'); if (response.ok) { this.currentUser = await response.json(); this.isLoggedIn = true; } } catch (error) { console.log('Not logged in'); } }, async login() { this.isLoading = true; this.loginError = ''; try { const response = await fetch('/api/users/login', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(this.loginForm) }); const data = await response.json(); if (response.ok) { this.currentUser = data.user; this.isLoggedIn = true; this.loginForm = { email: '', password: '' }; // Redirect based on user role if (data.user.role === 'owner') { window.location.href = 'owner-dashboard.html'; } else if (data.user.role === 'walker') { window.location.href = 'walker-dashboard.html'; } } else { this.loginError = data.error || 'Login failed'; } } catch (error) { this.loginError = 'An error occurred during login'; } finally { this.isLoading = false; } } } }).mount('#app');
